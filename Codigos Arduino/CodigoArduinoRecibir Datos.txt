const int ledPin = 13;

void setup() {
  pinMode(ledPin, OUTPUT);
  Serial.begin(115200);
}

void loop() {
  if (Serial.available()) {
    char comando = Serial.read();
    if (comando == 'L') {
      digitalWrite(ledPin, HIGH);
    } else if (comando == 'O') {
      digitalWrite(ledPin, LOW);
    }
  }
}
