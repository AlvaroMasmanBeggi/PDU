void lectura_tensionAC(){
    lectura = 0;
    valor_inst = 0;
    for (int i = 0; i < numero_muestrasAC; i++){
           lectura = analogRead(A0);
           valor_inst = lectura * lectura + valor_inst;

     }
    valor_inst /= numero_muestrasAC;
    volt_value = sqrt(valor_inst);
    Serial.println(volt_value);
    delay(10);
}
